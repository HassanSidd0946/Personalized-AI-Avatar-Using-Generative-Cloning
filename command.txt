Exact Commands:
# Navigate into the Wav2Lip directory
cd Wav2Lip

# Create the required directories
mkdir -p checkpoints
mkdir -p face_detection/detection/sfd

# Now download the weights
# 1. Face detection model
Invoke-WebRequest -Uri "https://www.adrianbulat.com/downloads/python-fan/s3fd-619a316812.pth" -OutFile "face_detection/detection/sfd/s3fd.pth"

# 2. GFPGAN model
Invoke-WebRequest -Uri "https://github.com/TencentARC/GFPGAN/releases/download/v1.3.0/GFPGANv1.3.pth" -OutFile "checkpoints/GFPGANv1.3.pth"

# 3. Wav2Lip model (this one is larger, ~150MB)
Invoke-WebRequest -Uri "https://huggingface.co/camenduru/Wav2Lip/resolve/main/checkpoints/wav2lip_gan.pth" -OutFile "checkpoints/wav2lip_gan.pth"

# Go back to parent directory
cd ..
